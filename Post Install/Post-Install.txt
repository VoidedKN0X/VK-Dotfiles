# Install OhMyZSH
- sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"

# Configure Greetd
- sudo nano /etc/greetd/config.toml (tuigreet --cmd start-hyprland)

# Edit Bootloader
- sudo nano /boot/loader/loader.conf
- sudo nano /boot/loader/entries/...

# Edit pam.d greetd
- sudo nano /etc/pam.d/greetd
- Add "auth       optional     pam_gnome_keyring.so" and "session    optional     pam_gnome_keyring.so auto_start"

# Add TrueNAS to fstab
# TrueNAS
192.168.1.9:/mnt/Data/axel    /media/TrueNAS  nfs     noauto,user,_netdev,bg  0       0

# Generate yubikey keys
- pamu2fcfg > ~/.config/Yubico/u2f_keys

# Enroll yubikey for LUKS
- sudo systemd-cryptenroll --fido2-device=auto /dev/xxx

# Add "auth sufficient pam_u2f.so authfile=~/.config/Yubico/u2f_keys cue" to
- sudo nano /etc/pam.d/system-login
- sudo nano /etc/pam.d/system-auth
- sudo nano /etc/pam.d/greetd
- sudo nano /etc/pam.d/sudo

#Edit "/etc/kernel/cmdline" and add plymouth to HOOKS before encrypt and after systemd

# Edit "/etc/mkinitcpio.d/..." to remove splash

- sudo mkinitcpio -P

# Secure Boot
- sudo -i
- sbctl create-keys
- sbctl enroll-keys -m
- sbctl sign -s -o /usr/lib/systemd/boot/efi/systemd-bootx64.efi.signed /usr/lib/systemd/boot/efi/systemd-bootx64.efi
- sbctl sign -s /boot/vmlinuz-linux-zen
- bootctl install
- sbctl verify

# Fix Logitech mouse
- sudo nano /etc/modprobe.d/hid-logitech.conf
- blacklist hid_logitech_hidpp
- blacklist hid_logitech_dj
- sudo mkinitcpio -P
